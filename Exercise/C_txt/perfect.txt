#include<stdio.h>

int main()
{
	int n;
	printf("Enter the Number of first perfect numbers: ");
	scanf("%d",&n);
	if(n==1)
	{
		printf("%d",1);

	}else
	{
		printf("%d\t",1);
		unsigned int i,j,sum=0;
		for(i=1,j=2;i<n;j++)
		{	
			unsigned int k=1;
			while(k<=j/2)
			{
				//printf("%d %d",k,j);
				if(j%k==0)
				{
					sum=sum+k;
					//printf("%d\t",sum);
				}
				k++;
			}
			if(sum==j)
			{
				printf("%d\t",j);
				i++;
			}
			sum=0;
		}
	}
}
