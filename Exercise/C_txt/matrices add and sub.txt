#include<stdio.h>
#include<conio.h>
main()
{
int i,j,m,n,p,q,a[100][100],b[100][100],c[100][100],s[100][100];
printf("Enter the order of first matrix :\n");
scanf("%d%d",&m,&n);
printf("Enter the order of second matrix: \n");
scanf("%d%d",&p,&q);
if(m!=p||n!=q)
	{
		printf("The addition matrix is no possible");
	}
else
	{
		printf("Enter the first matrix:\n");
			for(i=0;i<m;i++)
			for(j=0;j<n;j++)
			scanf("%d",&a[i][j]);
		printf("Enter the second matrix:\n");
		for(i=0;i<p;i++)
		for(j=0;j<q;j++)
		scanf("%d",&b[i][j]);
		for(i=0;i<m;i++)
		for(j=0;j<q;j++)
		c[i][j]=a[i][j]+b[i][j];
		printf("Sum of the matrices:\n");	
		for(i=0;i<m;i++)
		{
        printf("\n");
        for(j=0;j<q;j++)
        printf("%d\t",c[i][j]);
		}
		for(i=0;i<m;i++)
		for(j=0;j<q;j++)
		s[i][j]=a[i][j]-b[i][j];
		printf("\nSubstraction of the matrices:\n");	
		for(i=0;i<m;i++)
		{
        printf("\n");
        for(j=0;j<q;j++)
        printf("%d\t",s[i][j]);
	}
	}
}
